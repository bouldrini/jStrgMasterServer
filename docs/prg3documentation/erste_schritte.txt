			Konstruktor
			boolean can_invite?(FileFolder)
				primär: Gibt true zurück wenn ein User das Privilege besitzt einen anderen User zu diesem FileFolder einzuladen

			boolean can_read?(File)
				primär: Gibt true zurück wenn ein User das Privilege besitzt ein File zu sehen
			boolean can_read?(FileFolder)
				primär: Gibt true zurück wenn ein User das Privilege besitzt einen FileFolder zu sehen

			boolean can_write?(File)
				primär: Gibt true zurück wenn ein User das Privilege besitzt ein File zu sehen
			boolean can_write?(FileFolder)
				primär: Gibt true zurück wenn ein User das Privilege besitzt einen FileFolder zu sehen

			boolean has_role?(Symbol)
				primär: Gibt true zurück wenn ein User die gegebene Role besitzt
			Role role()
				primär: Gibt die Role eines Users zurück
